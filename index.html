	public function store(CreateBannerRequest $request)
	{
	    $request =    $this->saveFiles($request);
	    if($request->title == null){
	      $request['title'] = 'no title';
        }
	    $request=new  Request(array_merge($request->all(),['sort'=>Banner::max('sort')+1]));
		$banner=      Banner::create($request->all());
		return $banner;
	}
B) Update function :


	public function update($id, UpdateBannerRequest $request)
	{
		$banner = Banner::findOrFail($id);

        $request = $this->saveFiles($request);

        $banner->update($request->all());

        \Session::flash('success', 'edited successfully');

        return redirect()->route(config('quickadmin.route').'.banner.index');
	}
c) Delete function :


	public function destroy($id)
	{
		Banner::destroy($id);

		return 1;
	}

       protected $fillable = ['title'];

    public $relation_ids = [];

    public function menus()
    {
        return $this->belongsToMany(Menu::class);
    }

    public function canAccessMenu($menu)
    {
        if ($menu instanceof Menu) {
            $menu = $menu->id;
        }

        if (! isset($this->relation_ids['menus'])) {
            $this->relation_ids['menus'] = $this->menus()->pluck('id')->flip()->all();
        }

        return isset($this->relation_ids['menus'][$menu]);
    }

     * The attributes that are mass assignable.
